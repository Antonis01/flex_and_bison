<root> ::= <LinearLayout> | <RelativeLayout>

<LinearLayout> ::= "<LinearLayout>" <content> <LinearAttributes> <LinearLayout> "</LinearLayout>"
<RelativeLayout> ::= "<RelativeLayout>" <content>* <RelativeAttributes> <RelativeLayout> "</RelativeLayout>"

<content> ::= <element> | <root>  
<element> ::= <TextView/> | <ImageView/> | <Button/> | <RadioGroup/> | <ProgressBar/> | <element> <content>

<LinearAttributes> ::= <PrimaryAttributes> <AndroidID>* <AndroidOrientation>* 
<RelativeAttributes> ::= <PrimaryAttributes> <AndrtoidID>* 

<TextView/> ::= "<TextView>" <PrimaryAttributes> <AndroidText> <AndroidID>* <AndroidTextColour>* "</TextView>"
<ImageView/> ::= "<ImageView>" <PrimaryAttributes> <AndroidSRC> <AndroidID>* <AndroidPadding>* "</ImageView>"
<Button/> ::= "<Button>" <PrimaryAttributes> <AndroidText> <AndroidID>* <AndroidTextColour>* "</Button>"
<RadioGroup/> ::= "<RadioGroup>" <RadioButton/> <AndroidID>* <AndroidCheckedButton>* "</RadioGroup>"
<RadioButton/> ::= "<RadioButton>" <PrimaryAttributes> <AndroidText> <AndroidID>* "</RadioButton>"
<ProgressBar/> ::= "<ProgressBar>" <PrimaryAttributes> <AndroidID>* <AndroidMax>* <AndroidProgress>* "</ProgressBar>"

<PrimaryAttributes> ::= <AndroidHeight> <AndroidWidth>

<AndroidHeight> ::= 'android:layout_height' '=' <positive_integer>
<AndroidWidth> ::= 'android:layout_width' '=' <positive_integer
<AndroidID> ::= 'android:id' '=' <positive_integer>
<AndroidOrientation> ::= 'android:orientation' '=' <alphanumeric>
<AndroidText> ::= 'android:text' '=' <alphanumeric>
<AndroidTextColour> ::= 'android:textColor' '=' <alphanumeric>
<AndroidSRC> ::= 'android:src' '=' <alphanumeric>
<AndroidPadding> ::= 'android:padding' '=' <positive_integer>
<AndroidCheckedButton> ::= 'android:checkedButton' '=' <alphanumeric>
<AndroidProgress> ::= 'android:progress' '=' <positive_integer>
<AndroidMax> ::= 'android:max' '=' <positive_integer>
